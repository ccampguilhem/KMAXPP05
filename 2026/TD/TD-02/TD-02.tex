\tdtitle{2}{Régression Logistique}

\begin{exercice}
On considère la fonction logistique: 

\begin{equation}
	\begin{aligned}
		&\sigma: \mathbb{R} \to ]0, 1[ \\
		&\sigma : t \mapsto \frac{1}{1 + e^{-t}}
	\end{aligned}
	\nonumber
\end{equation}

La fonction logistique est une fonction \textit{sigmoïde}. Littéralement, c'est une fonction avec une forme de \textit{S}. Mathématiquement, une fonction sigmoïde est croissante, continue et dérivable et admet deux asymptotes horizontales en $-\infty$ et $+\infty$

\begin{enumerate}
	\item Montrer que la dérivée de la fonction $\sigma(t)$ est $\sigma(t) \left(1 - \sigma(t)\right)$
	\item Montrer que $\sigma$ est strictement croissante sur $\mathbb{R}$.
	\item Montrer que $\lim_{t \rightarrow + \infty} \sigma(t) = 1$ et que $\lim_{t \rightarrow - \infty} \sigma(t) = 0$.
	\item Quelle autre fonction (trigonométrique) est également une sigmoïde ?
	\item Montrer que $\sigma$ possède la propriété suivante $\sigma(t) = 1 - \sigma(-t) \quad \forall t \in \mathbb{R}$.
	\item Déduire de la propriété précédente que $\sigma(0) = \frac{1}{2}$ et que $\sigma(t) \ge \frac{1}{2} \Longleftrightarrow t \ge 0$.
	\item Tracer approximativement la courbe de $\sigma$ sur $\mathbb{R}$.
\end{enumerate}
\end{exercice}

\begin{solution}
    \begin{enumerate}
        \item La fonction sigmoïde est définie par $\sigma(t) = (1 + e^{-t})^{-1}$. Calculons sa dérivée :
        \begin{align*}
            \sigma'(t) &= -1 \times (1 + e^{-t})^{-2} \times (-e^{-t}) \\
                       &= \frac{e^{-t}}{(1 + e^{-t})^2} \\
                       &= \frac{1}{1 + e^{-t}} \times \frac{e^{-t}}{1 + e^{-t}} \\
                       &= \sigma(t) \left( \frac{1 + e^{-t} - 1}{1 + e^{-t}} \right) \\
                       &= \sigma(t) (1 - \sigma(t))
        \end{align*}

        \item On sait que $\sigma(t) \in ]0, 1[$, donc $\sigma(t) > 0$ et $1 - \sigma(t) > 0$.
        Par conséquent, $\sigma'(t) > 0$ pour tout $t \in \mathbb{R}$. La fonction est donc strictement croissante.

        \item Limites :
        \begin{itemize}
            \item Quand $t \to +\infty$, $e^{-t} \to 0$, donc $\sigma(t) \to \frac{1}{1+0} = 1$.
            \item Quand $t \to -\infty$, $e^{-t} \to +\infty$, donc le dénominateur tend vers l'infini et $\sigma(t) \to 0$.
        \end{itemize}

        \item La fonction tangente hyperbolique ($\tanh$) est également une sigmoïde (elle prend ses valeurs dans $]-1, 1[$).

        \item Calculons $1 - \sigma(-t)$ :
        \[
            1 - \frac{1}{1 + e^{t}} = \frac{1 + e^t - 1}{1 + e^t} = \frac{e^t}{1 + e^t}
        \]
        En multipliant numérateur et dénominateur par $e^{-t}$ :
        \[
            \frac{e^t e^{-t}}{(1 + e^t)e^{-t}} = \frac{1}{e^{-t} + 1} = \sigma(t)
        \]
        La propriété est vérifiée.

        \item En utilisant la propriété précédente avec $t=0$ :
        \[ \sigma(0) = 1 - \sigma(-0) \implies 2\sigma(0) = 1 \implies \sigma(0) = \frac{1}{2} \]
        Comme la fonction est strictement croissante, $\sigma(t) \ge \frac{1}{2} \iff t \ge 0$.

        \item \textit{(Le tracé doit montrer une courbe en S passant par $(0, 0.5)$, tendant vers 0 en $-\infty$ et 1 en $+\infty$)}.
    \end{enumerate}
\end{solution}

\begin{exercice}
On cherche à construire un classifieur $C$, basé sur une régression logistique, qui prend en entrée $x \in \mathbb{R}$ et qui le classifie en la classe $0$ ou la classe $1$. Ce classifieur va être paramétré par un vecteur $\beta \in \mathcal{M}_{2,1}(\mathbb{R})$ de composantes $\beta_0$ et $\beta_1$. La classe associée sera $1$ si $C_{\beta}(x) \ge 1/2$ et $0$ sinon. Le classifieur $C$ est défini comme suit:

\begin{equation}
	\begin{aligned}
		C_{\beta}(x) &= \sigma \left(\beta_0 + \beta_1 \, x\right) \\
		             &= \frac{1}{ 1 + \exp{\left(- \beta_0 - \beta_1 \, x\right)}}
	\end{aligned}
	\nonumber
\end{equation}

Les réponses de l'exercice 1 peuvent être utilisées pour les questions de cet exercice.

\begin{enumerate}
	\item Pour $\beta = \begin{pmatrix}0 \\ 1\end{pmatrix}$ quels $x$ sont classés en $0$ et quels $x$ sont classés en $1$ ?
	\item Même question pour $\beta = \begin{pmatrix}1 \\ -1\end{pmatrix}$
	\item Pour un $x \ne 0$ et $\beta_0$ fixés, lorsque $\beta_1$ devient assez grand ($\beta_1 \to + \infty$), comment est classé $x$ ? Même question lorsque $\beta_1 \to -\infty$.
\end{enumerate}
\end{exercice}

\begin{solution}
    \begin{enumerate}
        \item Pour $\beta = \begin{pmatrix}0 \\ 1\end{pmatrix}$, on a $C_{\beta}(x) = \sigma(0 + 1 \cdot x) = \sigma(x)$.
        La condition $C_{\beta}(x) \ge 1/2$ équivaut à $\sigma(x) \ge 1/2$, c'est-à-dire $x \ge 0$ (d'après l'exercice 1).
        \begin{itemize}
            \item Les $x \ge 0$ sont classés en $1$.
            \item Les $x < 0$ sont classés en $0$.
        \end{itemize}

        \item Pour $\beta = \begin{pmatrix}1 \\ -1\end{pmatrix}$, on a $C_{\beta}(x) = \sigma(1 - x)$.
        La condition $C_{\beta}(x) \ge 1/2$ équivaut à $1 - x \ge 0$, soit $x \le 1$.
        \begin{itemize}
            \item Les $x \le 1$ sont classés en $1$.
            \item Les $x > 1$ sont classés en $0$.
        \end{itemize}

        \item Analysons le comportement limite pour $x \ne 0$ :
        \begin{itemize}
            \item Si $\beta_1 \to +\infty$ :
                \begin{itemize}
                    \item Si $x > 0$, $\beta_0 + \beta_1 x \to +\infty \implies \sigma(\dots) \to 1$. Classé $1$.
                    \item Si $x < 0$, $\beta_0 + \beta_1 x \to -\infty \implies \sigma(\dots) \to 0$. Classé $0$.
                \end{itemize}
            \item Si $\beta_1 \to -\infty$ :
                \begin{itemize}
                    \item Si $x > 0$, $\beta_0 + \beta_1 x \to -\infty \implies \sigma(\dots) \to 0$. Classé $0$.
                    \item Si $x < 0$, $\beta_0 + \beta_1 x \to +\infty \implies \sigma(\dots) \to 1$. Classé $1$.
                \end{itemize}
        \end{itemize}
    \end{enumerate}
\end{solution}

\begin{exercice}
Nous disposons des données $\mathcal{D}$ suivantes pour des joueurs de rugby. L'espace des caractéristiques d'entrée est défini par le poids $x$ et noté $\mathcal{X} \subseteq \mathbb{R}$. L'espace des caractéristiques de sortie est défini par la position $y$ noté $\mathcal{Y} = \{0, 1\}$:

\begin{center}
	\begin{tabular}{|c|c|}
		\hline
		\textbf{Poids} (kg) & \textbf{Position} \\ \hline
		112 &  avant = 1       \\
		85  &  trois-quart = 0 \\
		135 &  avant = 1       \\
		92  &  trois-quart = 0 \\
		\hline
	\end{tabular}
\end{center}

On suppose que les $n = 4$ observations sont indépendantes. Nous cherchons à établir un classifieur basé sur un modèle de régression logistique de la forme suivante, avec $\beta_0 \in \mathbb{R}$ et $\beta_1 \in \mathbb{R}$:

\begin{equation}
	C_{\beta}(x) = 
	\begin{cases}
		0 \quad \forall x \quad  \displaystyle \frac{1}{1 + \exp{\left(- \beta_0 - \beta_1 \, x\right)}} < \frac{1}{2} \\[10pt]
		1 \quad \forall x \quad  \displaystyle \frac{1}{1 + \exp{\left(- \beta_0 - \beta_1 \, x\right)}} \ge \frac{1}{2}
	\end{cases}
	\nonumber
\end{equation}

On note $\beta \in \mathcal{M}_{2,1}(\mathbb{R})$ le vecteur:

\begin{equation}
	\beta = \begin{pmatrix}\beta_0 \\ \beta_1\end{pmatrix}
	\nonumber
\end{equation}

On associe à chaque observation $i$ une variable aléatoire $Y^{(i)}$ suivant une loi de Bernoulli de paramètre $p^{(i)} = \sigma(\tilde{x}^{(i)} \beta)$:

\begin{equation}
	\begin{cases}
		P(Y^{(i)} = 1) = p^{(i)} \\
		P(Y^{(i)} = 0) = 1 - p^{(i)}
	\end{cases}
	\nonumber
\end{equation}

Les réponses de l'exercice 1 peuvent être utilisées pour les questions de cet exercice.

\begin{enumerate}
	\item Quelle application $\Phi: \mathcal{X} \to \tilde{\mathcal{X}} \subseteq \mathbb{R}^2$ permet-elle de calculer la probabilité pour l'observation $i$ sous la forme $\sigma(\tilde{x}^{(i)} \beta)$ où $\sigma$ est la fonction logistique. Quelle est la matrice de design $\tilde{X}$ pour ce problème de classification ?
	\item Vérifier que la probabilité de la loi de Bernoulli pour la variable aléatoire $Y^{(i)}$ peut s'écrire $P(Y^{(i)} = y^{(i)}) = \left( p^{(i)} \right)^{y^{(i)}} \left(1 - p^{(i)}\right)^{\left(1 - y^{(i)}\right)}$.
	\item Pour une valeur fixe de $\beta$, quelle est la vraisemblance sur $\mathcal{D}$ selon les variables aléatoires de Bernoulli introduites ci-dessus ?
	\item Calculer la log-vraisemblance négative moyenne sur $\mathcal{D}$. Sous quelle autre dénomination usuelle cette quantité est-elle aussi connue ?
	\item Quelle condition doit satisfaire $\beta$ pour être la meilleure estimation de ce problème de régression logistique ?
	\item On rappelle que $\tilde{x}^{(i)}$ est le vecteur ligne de l'observation $i$ dans l'espace augmenté $\tilde{\mathcal{X}}$. On pose ${t^{(i)} = \tilde{x}^{(i)} \beta = \beta_0 + \beta_1 x^{(i)}}$. Démontrer que $\displaystyle \frac{d t^{(i)}}{d\beta} = \tilde{x}^{(i)T}$.
	\item Démontrer que $\displaystyle \frac{d}{d t^{(i)}}  \ln{\left[\sigma\left(t^{(i)}\right)\right]} = 1 - \sigma\left(t^{(i)}\right)$. 
	\item En déduire que $\displaystyle \frac{d}{d \beta}  \ln{\left[\sigma\left(\tilde{x}^{(i)} \beta\right)\right]} = \left[1 - \sigma\left(\tilde{x}^{(i)} \beta\right) \right] \tilde{x}^{(i)T}$.
	\item Démontrer que $\displaystyle \frac{d}{d t^{(i)}}  \ln{\left[1 - \sigma\left(t^{(i)}\right)\right]} = - \sigma\left(t^{(i)}\right)$.
	\item En déduire que $\displaystyle \frac{d}{d \beta}  \ln{\left[1 - \sigma\left(\tilde{x}^{(i)} \beta\right)\right]} = - \sigma\left(\tilde{x}^{(i)} \beta\right) \tilde{x}^{(i)T}$.
	\item Démontrer que le gradient de la log-vraisemblance négative moyenne par rapport à $\beta$ est égal à $\displaystyle -\frac{1}{n} \sum_{i=1}^n \left( y^{(i)} - \sigma(\tilde{x}^{(i)} \beta)\right) \tilde{x}^{(i)T}$.
	\item Démontrer que ce même gradient peut s'écrire sous la forme suivante: $\displaystyle -\frac{1}{n} \tilde{X}^T \left( Y - \sigma(\tilde{X} \beta)\right)$.
	\item On fixe $\beta = \begin{pmatrix}-90 \\ 1\end{pmatrix}$. On pourrait calculer que le gradient de la log-vraisemblance négative moyenne est à peu près égal à $\begin{pmatrix}0,22 \\ 20,4\end{pmatrix}$. Que déduire de cette valeur de $\beta$ ?
	\item En gardant la même valeur de $\beta$, quel est le poids à partir duquel le classifieur va prédire une classe $1$ ? En gardant $\beta_1 = 1$, quelle serait une meilleure alternative pour $\beta_0$ afin d'obtenir un meilleur classifieur ? Justifier votre choix pour cette alternative.
	\item La figure ci-dessous permet de visualiser le logarithme base 10 de la norme du gradient. Le $\beta = \begin{pmatrix}-90 \\ 1\end{pmatrix}$ de la question précédente est représenté. En prenant $\beta_1 = 1.0$ quelle valeur de $\beta_0$ permet de minimiser le gradient ? Quelle information cette visualisation du gradient apporte-t-elle?
	\begin{center}
		\includegraphics[width=0.7\textwidth]{gradient.png}
	\end{center}
\end{enumerate}
\end{exercice}

\begin{solution}
    \begin{enumerate}
        \item L'application est $\Phi(x) = (1, x)$. La matrice de design $\tilde{X}$ est obtenue en empilant les vecteurs lignes augmentés :
        \[
            \tilde{X} = \begin{pmatrix}
                1 & 112 \\
                1 & 85 \\
                1 & 135 \\
                1 & 92
            \end{pmatrix}
        \]

        \item Vérifions les deux cas possibles pour $y^{(i)} \in \{0, 1\}$ :
        \begin{itemize}
            \item Si $y^{(i)} = 1$, la formule donne $(p^{(i)})^1 (1-p^{(i)})^0 = p^{(i)}$, ce qui est bien $P(Y^{(i)}=1)$.
            \item Si $y^{(i)} = 0$, la formule donne $(p^{(i)})^0 (1-p^{(i)})^1 = 1-p^{(i)}$, ce qui est bien $P(Y^{(i)}=0)$.
        \end{itemize}

        \item Les observations étant indépendantes, la vraisemblance $\mathcal{L}(\beta)$ est le produit des probabilités individuelles :
        \[
            \mathcal{L}(\beta) = \prod_{i=1}^n P(Y^{(i)} = y^{(i)}) = \prod_{i=1}^n \left( p^{(i)} \right)^{y^{(i)}} \left(1 - p^{(i)}\right)^{\left(1 - y^{(i)}\right)}
        \]

        \item La log-vraisemblance négative moyenne est :
        \[
            J(\beta) = - \frac{1}{n} \ln(\mathcal{L}(\beta)) = - \frac{1}{n} \sum_{i=1}^n \left[ y^{(i)} \ln(p^{(i)}) + (1 - y^{(i)}) \ln(1 - p^{(i)}) \right]
        \]
        Cette quantité est usuellement appelée l'\textbf{Entropie Croisée} (Cross-Entropy) ou Log-Loss.

        \item Pour obtenir la meilleure estimation, il faut maximiser la vraisemblance, ce qui revient à minimiser la log-vraisemblance négative $J(\beta)$. La condition nécessaire est que le gradient de $J(\beta)$ par rapport à $\beta$ soit nul : $\nabla_\beta J(\beta) = 0$.

        \item On a $t^{(i)} = \tilde{x}^{(i)} \beta$. Puisque la dépendance est linéaire, le gradient par rapport à $\beta$ est le vecteur des coefficients, c'est-à-dire le vecteur $\tilde{x}^{(i)}$ transposé (pour respecter les dimensions d'un vecteur colonne) :
        \[ \frac{d t^{(i)}}{d\beta} = \tilde{x}^{(i)T} \]

        \item Utilisons la dérivée composée et le résultat de l'exercice 1 ($\sigma' = \sigma(1-\sigma)$) :
        \[
            \frac{d}{d t} \ln(\sigma(t)) = \frac{\sigma'(t)}{\sigma(t)} = \frac{\sigma(t)(1-\sigma(t))}{\sigma(t)} = 1 - \sigma(t)
        \]

        \item Par composition (règle de la chaîne) :
        \[
            \frac{d}{d \beta} \ln(\sigma(\tilde{x}^{(i)} \beta)) = \frac{d \ln \sigma}{d t^{(i)}} \times \frac{d t^{(i)}}{d \beta} = (1 - \sigma(t^{(i)})) \tilde{x}^{(i)T}
        \]

        \item De même :
        \[
            \frac{d}{d t} \ln(1-\sigma(t)) = \frac{-\sigma'(t)}{1-\sigma(t)} = \frac{-\sigma(t)(1-\sigma(t))}{1-\sigma(t)} = -\sigma(t)
        \]

        \item Par composition :
        \[
            \frac{d}{d \beta} \ln(1-\sigma(\tilde{x}^{(i)} \beta)) = -\sigma(t^{(i)}) \tilde{x}^{(i)T}
        \]

        \item Calculons le gradient de $J(\beta)$ terme à terme :
        \begin{align*}
            \nabla_\beta J(\beta) &= - \frac{1}{n} \sum_{i=1}^n \left[ y^{(i)} \nabla_\beta \ln(p^{(i)}) + (1 - y^{(i)}) \nabla_\beta \ln(1 - p^{(i)}) \right] \\
            &= - \frac{1}{n} \sum_{i=1}^n \left[ y^{(i)} (1 - p^{(i)})\tilde{x}^{(i)T} - (1 - y^{(i)}) p^{(i)}\tilde{x}^{(i)T} \right] \\
            &= - \frac{1}{n} \sum_{i=1}^n \left[ (y^{(i)} - y^{(i)}p^{(i)} - p^{(i)} + y^{(i)}p^{(i)}) \tilde{x}^{(i)T} \right] \\
            &= - \frac{1}{n} \sum_{i=1}^n \left( y^{(i)} - p^{(i)} \right) \tilde{x}^{(i)T}
        \end{align*}
        Avec $p^{(i)} = \sigma(\tilde{x}^{(i)} \beta)$, on retrouve la formule demandée.

        \item Sous forme matricielle, le vecteur résidu est $(Y - \sigma(\tilde{X}\beta))$. La somme des produits par $\tilde{x}^{(i)T}$ correspond au produit matriciel $\tilde{X}^T$ par ce vecteur résidu.
        \[ \nabla_\beta J(\beta) = -\frac{1}{n} \tilde{X}^T \left( Y - \sigma(\tilde{X} \beta)\right) \]

        \item Le gradient n'est pas nul (sa norme est significative). Cela signifie que le vecteur $\beta$ actuel n'est pas l'optimum de la fonction de coût : on peut encore améliorer le classifieur.

        \item Frontière de décision : $\beta_0 + \beta_1 x = 0 \iff -90 + 1 \cdot x = 0 \iff x = 90$ kg.
        \begin{itemize}
            \item Le classifieur prédit 1 si $x \ge 90$.
            \item Dans les données, le joueur de 92 kg est un "trois-quart" (classe 0), mais est classé 1 (car $92 > 90$). C'est une erreur.
            \item Pour corriger cela, la frontière doit se situer entre le plus lourd des 0 (92 kg) et le plus léger des 1 (112 kg). Une bonne alternative est le milieu : $\frac{92 + 112}{2} = 102$ kg.
            \item Pour avoir une frontière à 102 avec $\beta_1 = 1$, il faut $x + \beta_0 = 0 \iff 102 + \beta_0 = 0 \iff \beta_0 = -102$.
        \end{itemize}

        \item La visualisation montre la magnitude du gradient. La zone la plus sombre (gradient faible) correspond aux meilleures valeurs de paramètres. Pour $\beta_1 = 1$, on voit que le minimum (vallée) se situe aux alentours de $\beta_0 \approx -100$ ou $-102$, ce qui confirme notre déduction précédente.
    \end{enumerate}
\end{solution}
